const quizzes = {

  6: [

    {q: "What is 5 + 3?", options: ["6", "8", "9"], answer: "8"},

    {q: "Which planet is known as the Red Planet?", options: ["Earth", "Mars", "Venus"], answer: "Mars"},

    {q: "Which is a mammal?", options: ["Shark", "Dolphin", "Frog"], answer: "Dolphin"},

    {q: "What is 10 - 4?", options: ["6", "5", "7"], answer: "6"},

    {q: "What color do you get by mixing red and white?", options: ["Pink", "Purple", "Orange"], answer: "Pink"},

    {q: "Which animal lays eggs?", options: ["Dog", "Cat", "Hen"], answer: "Hen"}

  ],

  // ... baki classes ka code yaha paste kar do (same as your code)

};

let studentName = "";

let selectedClass = "";

function startQuiz() {

  studentName = document.getElementById("studentName").value;

  selectedClass = document.getElementById("classSelect").value;

  if(studentName === ""){

    alert("Please enter your name!");

    return;

  }

  document.getElementById("loginSection").classList.add("hidden");

  document.getElementById("quizSection").classList.remove("hidden");

  document.getElementById("quizTitle").innerText = `Quiz for Class ${selectedClass}`;

  loadQuiz();

}

function loadQuiz() {

  const quizBox = document.getElementById("quizContent");

  quizBox.innerHTML = "";

  quizzes[selectedClass].forEach((item, index) => {

    let qDiv = document.createElement("div");

    qDiv.classList.add("question");

    qDiv.innerHTML = `<p>${index+1}. ${item.q}</p>`;

    item.options.forEach(opt => {

      qDiv.innerHTML += `<label><input type='radio' name='q${index}' value='${opt}'> ${opt}</label><br>`;

    });

    quizBox.appendChild(qDiv);

  });

}

function submitQuiz() {

  let score = 0;

  quizzes[selectedClass].forEach((item, index) => {

    const options = document.getElementsByName("q"+index);

    options.forEach(opt => {

      if(opt.checked && opt.value === item.answer){

        score++;

      }

    });

  });

  const total = quizzes[selectedClass].length;

  const percentage = (score / total) * 100;

  let grade = "";

  if(percentage >= 80) grade = "A";

  else if(percentage >= 60) grade = "B";

  else grade = "C";

  document.getElementById("quizSection").classList.add("hidden");

  document.getElementById("resultSection").classList.remove("hidden");

  document.getElementById("resultText").innerText = `${studentName} (Class ${selectedClass})\nScore: ${score}/${total}\nPercentage: ${percentage.toFixed(2)}%\nGrade: ${grade}`;

}

function goHome(){

  document.getElementById("resultSection").classList.add("hidden");

  document.getElementById("loginSection").classList.remove("hidden");

  document.getElementById("studentName").value = "";

}
